 31     " When editing a file, always jump to the last known cursor position.
 32     " Don't do it when the position is invalid, when inside an event handler
 33     " (happens when dropping a file on gvim) and for a commit message (it's
 34     " likely a different one than last time).
 35     autocmd BufReadPost *
 36                 \ if line("'\"") >= 1 && line("'\"") <= line("$") && &ft !~# 'commit'
 37                 \ |   exe "normal! g`\""
 38                 \ | endif

143 local nvim_lsp = require('lspconfig')

145 -- Use an on_attach function to only map the following keys
146 -- after the language server attaches to the current buffer
147 local on_attach = function(client, bufnr)
148   local function buf_set_keymap(...) vim.api.nvim_buf_set_keymap(bufnr, ...) end
149   local function buf_set_option(...) vim.api.nvim_buf_set_option(bufnr, ...) end
150 
151   --Enable completion triggered by <c-x><c-o>
152   buf_set_option('omnifunc', 'v:lua.vim.lsp.omnifunc')
153 
154   -- Mappings.
155   local opts = { noremap=true, silent=true }
156 
157   -- See `:help vim.lsp.*` for documentation on any of the below functions
158   buf_set_keymap('n', 'gD', '<Cmd>lua vim.lsp.buf.declaration()<CR>', opts)
159   buf_set_keymap('n', 'gd', '<Cmd>lua vim.lsp.buf.definition()<CR>', opts)
160   buf_set_keymap('n', 'K', '<Cmd>lua vim.lsp.buf.hover()<CR>', opts)
161   buf_set_keymap('n', 'gi', '<cmd>lua vim.lsp.buf.implementation()<CR>', opts)
162   buf_set_keymap('n', '<C-k>', '<cmd>lua vim.lsp.buf.signature_help()<CR>', opts)
163   buf_set_keymap('n', '<space>wa', '<cmd>lua vim.lsp.buf.add_workspace_folder()<CR>', opts)
164   buf_set_keymap('n', '<space>wr', '<cmd>lua vim.lsp.buf.remove_workspace_folder()<CR>', opts)
165   buf_set_keymap('n', '<space>wl', '<cmd>lua print(vim.inspect(vim.lsp.buf.list_workspace_folders()))<CR>', opts)
166   buf_set_keymap('n', '<space>D', '<cmd>lua vim.lsp.buf.type_definition()<CR>', opts)
167   buf_set_keymap('n', '<space>rn', '<cmd>lua vim.lsp.buf.rename()<CR>', opts)
168   buf_set_keymap('n', '<space>ca', '<cmd>lua vim.lsp.buf.code_action()<CR>', opts)
169   buf_set_keymap('n', 'gr', '<cmd>lua vim.lsp.buf.references()<CR>', opts)

172   buf_set_keymap('n', ']d', '<cmd>lua vim.lsp.diagnostic.goto_next()<CR>', opts)
